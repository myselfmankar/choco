public class BankAccountSystem {

    // 🔹 Add Account
    public static void addAccount(String accId, String name, String email, Date dob, String dept) {
        BankAccount__c acc = new BankAccount__c();

        acc.Name = accId; // Emp ID (Record Name)
        acc.Emp_Name__c = name;
        acc.Email__c = email;
        acc.Birth_Date__c = dob;
        acc.Department__c = dept;

        insert acc;
        System.debug('Account Added Successfully!');
    }

    // 🔹 View Accounts
    public static void viewAccounts() {
        List<BankAccount__c> accList = [
            SELECT Name, Emp_Name__c, Email__c, Birth_Date__c, Department__c 
            FROM BankAccount__c
        ];

        for(BankAccount__c acc : accList) {
            System.debug('Account ID: ' + acc.Name);
            System.debug('Name: ' + acc.Emp_Name__c);
            System.debug('Email: ' + acc.Email__c);
            System.debug('DOB: ' + acc.Birth_Date__c);
            System.debug('Department: ' + acc.Department__c);
            System.debug('----------------------');
        }
    }

    // 🔹 Update Account
    public static void updateAccount(String accId, String newEmail) {
        BankAccount__c acc = [
            SELECT Id, Email__c 
            FROM BankAccount__c 
            WHERE Name = :accId 
            LIMIT 1
        ];

        acc.Email__c = newEmail;
        update acc;

        System.debug('Account Updated Successfully!');
    }

    // 🔹 Delete Account
    public static void deleteAccount(String accId) {
        BankAccount__c acc = [
            SELECT Id 
            FROM BankAccount__c 
            WHERE Name = :accId 
            LIMIT 1
        ];

        delete acc;
        System.debug('Account Deleted Successfully!');
    }
}
